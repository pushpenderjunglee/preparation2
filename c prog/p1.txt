#include <stdio.h>

int main() {
    int i,j,k,l,row;
    printf("Enter row\n");
    scanf("%d",row);
    
for(i=1;i<=row;i++)
{
    for(j=row-1;j>=i;j--)
    printf(" ");

    for(k=0;k<i;k++)
    printf("#");
     
    for(l=1;l<i;l++)
    printf("#");
    
    printf("\n");
}
    
}
